# Solutionnaire Technique (Code de référence) - Système de Station Météo

### Classe CapteurMeteo.cs (Projet Metier)
```csharp
namespace StationMeteo.Metier
{
    public class CapteurMeteo
    {
        private string _nom;
        private double _temperature;
        private int _humidite;

        // Méthode de validation privée demandée
        private bool ContientDesChiffres(string texte)
        {
            foreach (char c in texte)
            {
                if (char.IsDigit(c)) return true;
            }
            return false;
        }

        public string Nom 
        { 
            get => _nom; 
            set 
            {
                if (string.IsNullOrWhiteSpace(value))
                    throw new ArgumentException("Le nom ne peut pas être vide.");
                
                if (ContientDesChiffres(value))
                    throw new ArgumentException("Le nom ne doit pas contenir de chiffres.");

                _nom = value;
            }
        }

        public double Temperature 
        { 
            get => _temperature; 
            set 
            {
                // Bornage (Clamping)
                if (value < -60.0) _temperature = -60.0;
                else if (value > 60.0) _temperature = 60.0;
                else _temperature = value;
            }
        }

        public int Humidite 
        { 
            get => _humidite; 
            set 
            {
                if (value < 0 || value > 100)
                    throw new ArgumentOutOfRangeException(nameof(Humidite), "L'humidité doit être entre 0 et 100.");
                _humidite = value;
            }
        }

        public CapteurMeteo(string nom) : this(nom, 0.0, 50) { }

        public CapteurMeteo(string nom, double temperature, int humidite)
        {
            // Passage par les propriétés pour valider
            this.Nom = nom;
            this.Temperature = temperature;
            this.Humidite = humidite;
        }

        public string LireDonnees()
        {
            return $"Capteur : {Nom} | Temp : {Temperature}°C | Humidité : {Humidite}%";
        }
    }
}
```

using StationMeteo.Metier;
